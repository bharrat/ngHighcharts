ngHighcharts

AngularJs directive for Highcharts

Download this project as a .zip file Download this project as a tar.gz file

Usage

Controller: 

function DemoCtrl($scope) {
    var chartData = {
        series: [{
            type: 'pie',
            name: 'Browser share',
            data: [
                ['Firefox',   45.0],
                ['IE',       26.8],
                {name: 'Chrome', y: 12.8, sliced: true,selected: true},
                ['Safari',    8.5],
                ['Opera',     6.2],
                ['Others',   0.7]
            ]
        }]
    };

View:

<div data-highchart data-options="chartData" width="300" height="400"></div>
or
<highchart data-options="chartData" width="300" height="400"></highchart>
